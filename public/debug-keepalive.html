🔍 Keep-Alive 调试助手

步骤 1: 复制以下代码到浏览器控制台
// === Keep-Alive 完整诊断 ===
const route = window.$route || (window.$nuxt?.$route)
const store = window.useTagsViewStore?.() || window.$nuxt?.$store?.state?.tagsView

console.group('🔍 Keep-Alive 完整诊断')

// 1. 当前路由信息
console.group('📍 当前路由')
console.log('route.name:', route?.name)
console.log('route.path:', route?.path)
console.log('route.meta:', route?.meta)
console.log('route.meta.keepAlive:', route?.meta?.keepAlive)
console.groupEnd()

// 2. 组件信息
console.group('🎨 组件信息')
const matched = route?.matched?.[route.matched.length - 1]
const component = matched?.components?.default
console.log('组件 __name:', component?.__name)
console.log('组件 name:', component?.name)
console.log('defineOptions name:', component?.name)
console.groupEnd()

// 3. cachedViews
console.group('📦 缓存状态')
if (store) {
    console.log('cachedViews:', store.cachedViews)
    console.log('visitedViews:', store.visitedViews)
    const isInCache = store.cachedViews?.includes(route?.name)
    console.log('当前路由在缓存中:', isInCache ? '✅ 是' : '❌ 否')
} else {
    console.error('❌ 找不到 tagsView store')
}
console.groupEnd()

// 4. 匹配检查
console.group('🔄 匹配检查')
const routeName = route?.name
const componentName = component?.__name || component?.name
console.log('路由 name:', routeName)
console.log('组件 name:', componentName)
if (routeName === componentName) {
    console.log('✅ 匹配成功！')
} else {
    console.error('❌ 不匹配！这是问题所在！')
    console.log('需要修改组件的 defineOptions({ name: "' + routeName + '" })')
}
console.groupEnd()

console.groupEnd()

常见问题和解决方案

❌ 路由 name 和组件 name 不匹配

→ 确保 defineOptions({ name }) 与路由 name 完全相同

❌ cachedViews 是空数组

→ 检查 route.meta.keepAlive 是否为 true

→ 检查 router.afterEach 是否调用了 addView

❌ 后端返回 meta.keepAlive = 0

→ 需要在后端数据库中设置 keepAlive = 1

❌ key 使用了 route.fullPath

→ 改为使用 route.name
